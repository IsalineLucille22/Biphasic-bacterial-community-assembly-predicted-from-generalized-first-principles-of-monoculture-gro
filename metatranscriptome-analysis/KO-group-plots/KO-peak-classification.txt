counts_list=dir('*log2*');

all_results=struct();

for jj=1:length(counts_list);

counts=readtable(counts_list(jj).name);

%genomes

genomes=unique(counts.strain);

genome_results=[];

%time categories

time_categories=[1:6];

%split by the genomes

tmp=annotations(contains(annotations.shortname,genomes),:);

%suppress the double target column in the counts file (CAUS24_2_168_5), which is replaced
%by CAUS24_2_168h_5

counts.CAUS24_2_168_5=[];

%suppress the order number column in the counts file (Var1)

counts.Var1=[];

%collect individual values

for i=1:length(KO_list)

tmp_KO=tmp(contains(tmp.KEGG_Pathway,KO_list{i}),:);

%then find this group in the counts

tmp_counts=counts(ismember(counts.gene_id,tmp_KO.locus_tag),:);

sample_times=unique(tmp_counts.Properties.VariableNames(contains(tmp_counts.Properties.VariableNames,'CAUS')))';
sample_times=split(sample_times,'_');
sample_times=unique(sample_times(:,3));

%be careful, the order of sample times is 
%'12h'
%'168h'
%'22h'
%'38h'
%'3w'
%'70h'
%but the order in displays should be [1,3,4,6,2,5];

tmp_order=[1,3,4,6,2,5];

tmp_results=[];

for k=1:length(tmp_order)

tmp_time=tmp_counts(:,contains(tmp_counts.Properties.VariableNames,sample_times(tmp_order(k))));

tmp_replicate=table2array(tmp_time);
tmp_replicate(tmp_replicate==-1)=nan;

%because this is log2, we first have to unlog then sum, then take again log2
 
tmp_replicate=power(2,tmp_replicate);

tmp_replicate=nansum(tmp_replicate);
tmp_replicate=log2(tmp_replicate);

tmp_results(:,k)=tmp_replicate; %columns as the time points, rows as the replicate values

%replace again all -Inf as nan

tmp_results(tmp_results==-Inf)=nan;

end

genome_results{i}=tmp_results;

end

%make valid field_names

genomes = matlab.lang.makeValidName(genomes);

all_results.(string(genomes)).KO_expression=genome_results';

end



%%%%%%%%%%
%calculate the Pearson and the Spearman coefficients for all individual data
%and score the expression behaviour

names=fieldnames(all_results)


for jj=1:length(names)

score=[];

for i=1:length(all_results.(names{jj}).KO_expression)

data=nanmean(all_results.(names{jj}).KO_expression{i}',2);

[rho,pval]=corr(time_categories',data,'Type','Pearson');

if rho<-0.5 & pval<0.05

score(i) = 3; %monotonous down

elseif rho>0.5 & pval<0.05

score(i) = 4; %monotonous up

elseif any(data(2:3)>data(1)+1.2) && any(data(2:3) > data(5:6)+1.2)

score(i) = 1; %has a peak at tp2-3

elseif any(data(2:3)+1.2<data(1)) && any(data(2:3)+1.2< data(5:6))

score(i) = 2; %has a trough at tp2-3

elseif any(data(4)>data(1:3)+1.2) && any(data(4) > data(5:6)+1.2)

score(i) = 5; %has a peak at tp4

elseif any(data(4)+1.2<data(1:3)) && any(data(4)+1.2< data(5:6))

score(i) = 6; %has a trough at tp4

else

score(i) = 7;

end

end

all_results.(names{jj}).scores=score;

end

%%%try to correlate all profiles to each other across selected strains to find groups

names=fieldnames(all_results)

selected_names=[3,8,9,11,16,17,18,21];

%3=Caulobacter
%8=Devosia
%9=Flavobacterium
%11=Lysobacter

%combine all data - not used

end


score_rho=[];
score_pval=[];

for jj=1:length(all_data)
for kk=1:length(all_data)

xdata=nanmean(all_data{jj}',2);
ydata=nanmean(all_data{kk}',2);

[rho,pval]=corr(xdata,ydata,'Type','Pearson');

score_rho(jj,kk)=rho;
score_pval(jj,kk)=pval;

end
end

score_rho(score_rho<0)=nan;

dissimilarityMatrix = 1 - score_rho;
dissimilarityMatrix(dissimilarityMatrix<1e-6)=0;
dissimilarityMatrix(isnan(dissimilarityMatrix))=0;

%visualize clusters - not used

ranksum005=bsxfun(@lt,score_pval,0.05);

%do logic filtering but maintain the array size!

ranksum_threshold = nan(size(score_rho));

ranksum_threshold(ranksum005)=dissimilarityMatrix(ranksum005);

%only keep the positive correlations

%ranksum_threshold(ranksum_threshold<0)=nan;

ranksum_threshold(isnan(ranksum_threshold))=0;

Z = linkage(squareform(dissimilarityMatrix), 'complete');

% Step 3: Determine how many clusters to form
% Use 'cluster' function, set the distance threshold or the number of clusters
% Here we use a threshold example:
maxDistance = 0.995; % Set a threshold for cluster merging
clusters = cluster(Z, 'cutoff', maxDistance, 'criterion', 'distance');

% Step 4: Count the number of unique clusters
numClusters = length(unique(clusters));

% Step 5: (Optional) Visualize the dendrogram
figure;
dendrogram(Z);
title('Hierarchical Clustering Dendrogram');
xlabel('Sample Index');
ylabel('Distance');


%clustergram(ranksum_threshold); %not used


%try to group with kmeans

%[idx,C,sumd,D] = kmeans(ranksum_threshold,7,'MaxIter',10000,...
    'Display','final','Replicates',10);

KO_expression_groups=struct()

for i=1:numClusters

KO_expression_groups(i).group=find(clusters==i);

end


%correct for the number of the genome which it came from (i.e., multiple of 92)

for i=1:numClusters

tmp=KO_expression_groups(i).group;

for j=1:length(tmp)

if tmp(j,1)<=92

KO_expression_groups(i).group(j,2)=1;
KO_expression_groups(i).group(j,3)=tmp(j,1);

elseif 92<tmp(j,1) & tmp(j,1)<=2*92

KO_expression_groups(i).group(j,2)=2;
KO_expression_groups(i).group(j,3)=tmp(j,1)-1*92;

elseif 2*92<tmp(j,1) & tmp(j,1)<=3*92

KO_expression_groups(i).group(j,2)=3;
KO_expression_groups(i).group(j,3)=tmp(j,1)-2*92;

elseif 3*92<tmp(j,1) & tmp(j,1)<=4*92

KO_expression_groups(i).group(j,2)=4;
KO_expression_groups(i).group(j,3)=tmp(j,1)-3*92;

elseif 4*92<tmp(j,1) & tmp(j,1)<=5*92

KO_expression_groups(i).group(j,2)=5;
KO_expression_groups(i).group(j,3)=tmp(j,1)-4*92;

elseif 5*92<tmp(j,1) & tmp(j,1)<=6*92

KO_expression_groups(i).group(j,2)=6;
KO_expression_groups(i).group(j,3)=tmp(j,1)-5*92;

elseif 6*92<tmp(j,1) & tmp(j,1)<=7*92

KO_expression_groups(i).group(j,2)=7;
KO_expression_groups(i).group(j,3)=tmp(j,1)-6*92;

else

KO_expression_groups(i).group(j,2)=8;
KO_expression_groups(i).group(j,3)=tmp(j,1)-7*92;

end
end
