% Pvalues, FDR, volcano plot
g1=[20:24]; %tp 38 h
g2=[15:19]; %tp 22 h


RNA_seq=[lysobacter(:,g1) lysobacter(:,g2)];

%keep only rows where there are fewer values= -1 than 2 (so we can keep triplicates)

tmp=bsxfun(@eq, table2array(RNA_seq), -1);
idx=find(sum(tmp,2)<3);
RNA_seq=RNA_seq(idx,:);

%replace -1 by nan

RNA_seq=table2array(RNA_seq);
RNA_seq(RNA_seq==-1)=nan;

%fill missing values with the median for the mattest

for i=1:length(RNA_seq)

X(i,:)=fillmissing(RNA_seq(i,6:10),'movmedian',5);
Y(i,:)=fillmissing(RNA_seq(i,1:5),'movmedian',5);

end


PValues=mattest (Y,X,'permute',1000); % Perform two-sample t-test to evaluate differential expression of genes from two experimental conditions

FDR = mafdr(PValues, 'BHFDR', true); % Estimate positive false discovery rate for multiple hypothesis testing

SigStructure=mavolcanoplot(Y,X,FDR,'LogTrans',false,'Foldchange',2, 'Labels',lysobacter.gene_id(idx));
